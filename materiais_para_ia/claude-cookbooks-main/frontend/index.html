from neuralforge import Agent, Tool

# Crie agentes autônomos em 3 linhas
agent = Agent(
    model="claude-4-opus",
    tools=[Tool.web_search(), Tool.code_exec()],
    memory=True
)

result = await agent.run(
    "Analise nossos dados de vendas Q4 e
     gere um relatório executivo com insights"
)

print(result.report)  # ✨ Relatório completo
print(result.charts)  # 📊 Gráficos gerados
